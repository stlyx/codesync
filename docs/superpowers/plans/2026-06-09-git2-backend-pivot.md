# Rust git2 Backend Pivot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the in-progress gix backend plan with a git2/libgit2 backend that supports fetch and push without invoking the `git` binary.

**Architecture:** Keep the already-reviewed `sync::GitBackend` trait and orchestration, but implement its concrete backend with `git2::Repository`, transient remotes, libgit2 credentials callbacks, and non-force refspecs. Remove the `gix` dependency entirely so the final project uses Rust + git2/libgit2 for Git operations, with no Python and no `git` subprocesses.

**Tech Stack:** Rust 2024, `git2 0.21`, existing `clap`, `serde`, `serde_json`, `tiny_http`, `fs4`, `hmac`, `sha2`, `subtle`, `thiserror`, `tracing`, `tracing-subscriber`, `tempfile`.

```rust
use crate::config::{AppConfig, RemoteConfig};
use crate::credentials::ResolvedCredentials;
use crate::error::{CodeSyncError, Result};
use crate::sync::GitBackend;

pub struct Git2Backend {
    repo: Option<git2::Repository>,
}

impl Git2Backend {
    pub fn new() -> Self {
        Self { repo: None }
    }

    fn repo(&self) -> Result<&git2::Repository> {
        self.repo
            .as_ref()
            .ok_or_else(|| CodeSyncError::GitBackend("repository is not initialized".to_string()))
    }
}

impl Default for Git2Backend {
    fn default() -> Self {
        Self::new()
    }
}

pub fn branch_ref(branch: &str) -> String {
    format!("refs/heads/{branch}")
}

pub fn remote_branch_ref(remote: &str, branch: &str) -> String {
    format!("refs/remotes/{remote}/{branch}")
}

pub fn branch_fetch_refspec(remote: &str, branch: &str) -> String {
    format!("+refs/heads/{branch}:{}", remote_branch_ref(remote, branch))
}

pub fn tag_fetch_refspec() -> &'static str {
    "refs/tags/*:refs/tags/*"
}

pub fn branch_push_refspec(branch: &str) -> String {
    format!("refs/heads/{branch}:refs/heads/{branch}")
}

pub fn tag_push_refspec() -> &'static str {
    "refs/tags/*:refs/tags/*"
}

fn git_error(error: git2::Error) -> CodeSyncError {
    CodeSyncError::GitBackend(error.message().to_string())
}
```

```rust
impl GitBackend for Git2Backend {
    fn ensure_repo(&mut self, config: &AppConfig) -> Result<()> {
        if self.repo.is_some() {
            return Ok(());
        }
        if let Some(parent) = config.repo_dir.parent().filter(|parent| !parent.as_os_str().is_empty()) {
            std::fs::create_dir_all(parent).map_err(|err| CodeSyncError::io_error(parent, err))?;
        }
        let repo = if config.repo_dir.join("HEAD").exists() {
            git2::Repository::open_bare(&config.repo_dir).map_err(git_error)?
        } else {
            git2::Repository::init_bare(&config.repo_dir).map_err(git_error)?
        };
        self.repo = Some(repo);
        Ok(())
    }

    fn fetch_remote(&mut self, remote: &RemoteConfig, branch: &str) -> Result<String> {
        fetch_remote_impl(self.repo()?, remote, branch)
    }

    fn fetch_tags(&mut self, remote: &RemoteConfig) -> Result<()> {
        fetch_tags_impl(self.repo()?, remote)
    }

    fn local_branch_tip(&self, branch: &str) -> Result<Option<String>> {
        let refname = branch_ref(branch);
        match self.repo()?.refname_to_id(&refname) {
            Ok(oid) => Ok(Some(oid.to_string())),
            Err(err) if err.code() == git2::ErrorCode::NotFound => Ok(None),
            Err(err) => Err(git_error(err)),
        }
    }

    fn is_ancestor(&self, older: &str, newer: &str) -> Result<bool> {
        let older = git2::Oid::from_str(older).map_err(git_error)?;
        let newer = git2::Oid::from_str(newer).map_err(git_error)?;
        self.repo()?.graph_descendant_of(newer, older).map_err(git_error)
    }

    fn update_local_branch(&mut self, branch: &str, target: &str) -> Result<()> {
        let target = git2::Oid::from_str(target).map_err(git_error)?;
        self.repo()?
            .reference(&branch_ref(branch), target, true, "codesync fast-forward")
            .map_err(git_error)?;
        Ok(())
    }

    fn push_remote(&mut self, remote: &RemoteConfig, branch: &str, _target: &str) -> Result<()> {
        push_impl(self.repo()?, remote, &[branch_push_refspec(branch)])
    }

    fn push_tags(&mut self, remote: &RemoteConfig) -> Result<()> {
        push_impl(self.repo()?, remote, &[tag_push_refspec().to_string()])
    }
}
```

```rust
fn callbacks(remote: &RemoteConfig) -> Result<git2::RemoteCallbacks<'_>> {
    let credentials = ResolvedCredentials::from_config(&remote.credential)?;
    let mut callbacks = git2::RemoteCallbacks::new();
    callbacks.credentials(move |_url, username_from_url, allowed| {
        let Some(credentials) = credentials.as_ref() else {
            return Err(git2::Error::from_str("credentials are not configured"));
        };
        if allowed.contains(git2::CredentialType::USER_PASS_PLAINTEXT) {
            git2::Cred::userpass_plaintext(&credentials.username, &credentials.password)
        } else if allowed.contains(git2::CredentialType::USERNAME) {
            git2::Cred::username(username_from_url.unwrap_or(&credentials.username))
        } else {
            Err(git2::Error::from_str("unsupported credential type requested by remote"))
        }
    });
    Ok(callbacks)
}

fn fetch_options(remote: &RemoteConfig) -> Result<git2::FetchOptions<'_>> {
    let mut options = git2::FetchOptions::new();
    options.remote_callbacks(callbacks(remote)?);
    Ok(options)
}

fn push_options(remote: &RemoteConfig) -> Result<git2::PushOptions<'_>> {
    let mut options = git2::PushOptions::new();
    options.remote_callbacks(callbacks(remote)?);
    Ok(options)
}

fn with_anonymous_remote<T>(repo: &git2::Repository, remote: &RemoteConfig, f: impl FnOnce(&mut git2::Remote<'_>) -> Result<T>) -> Result<T> {
    let mut handle = repo.remote_anonymous(&remote.url).map_err(git_error)?;
    f(&mut handle)
}

fn fetch_remote_impl(repo: &git2::Repository, remote: &RemoteConfig, branch: &str) -> Result<String> {
    let refspec = branch_fetch_refspec(&remote.name, branch);
    let mut options = fetch_options(remote)?;
    with_anonymous_remote(repo, remote, |handle| {
        handle.fetch(&[refspec.as_str()], Some(&mut options), Some("codesync fetch branch")).map_err(git_error)?;
        let oid = repo.refname_to_id(&remote_branch_ref(&remote.name, branch)).map_err(git_error)?;
        Ok(oid.to_string())
    })
}

fn fetch_tags_impl(repo: &git2::Repository, remote: &RemoteConfig) -> Result<()> {
    let mut options = fetch_options(remote)?;
    with_anonymous_remote(repo, remote, |handle| {
        handle.fetch(&[tag_fetch_refspec()], Some(&mut options), Some("codesync fetch tags")).map_err(git_error)?;
        Ok(())
    })
}

fn push_impl(repo: &git2::Repository, remote: &RemoteConfig, refspecs: &[String]) -> Result<()> {
    let mut options = push_options(remote)?;
    with_anonymous_remote(repo, remote, |handle| {
        let refs = refspecs.iter().map(String::as_str).collect::<Vec<_>>();
        handle.push(&refs, Some(&mut options)).map_err(git_error)?;
        Ok(())
    })
}
```
